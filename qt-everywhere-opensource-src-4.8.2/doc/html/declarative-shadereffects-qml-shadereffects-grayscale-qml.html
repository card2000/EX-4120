 import QtQuick 1.0
 import Qt.labs.shaders 1.0

 Item {
     id: main
     anchors.fill: parent

     GrayscaleEffect {
         id: layer
         anchors.fill: parent

         source: ShaderEffectSource {
             sourceItem: Image { source: "images/desaturate.jpg" }
             live: false
             hideSource: true
         }

         SequentialAnimation on ratio {
             id: ratioAnimation
             running: true
             loops: Animation.Infinite
             NumberAnimation {
                 easing.type: Easing.Linear
                 to: 0.0
                 duration: 1500
             }
             PauseAnimation { duration: 1000 }
             NumberAnimation {
                 easing.type: Easing.Linear
                 to: 1.0
                 duration: 1500
             }
             PauseAnimation { duration: 1000 }
         }
     }
 }
